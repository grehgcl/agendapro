// Banco de dados principal (controla todas as barbearias)
const mainDb = new sqlite3.Database('master.db');

// Criar tabela de barbearias no banco master
mainDb.serialize(() => {
    mainDb.run(`CREATE TABLE IF NOT EXISTS barbearias (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    nome TEXT NOT NULL,
    cnpj TEXT,
    telefone TEXT,
    email TEXT UNIQUE NOT NULL,
    senha TEXT NOT NULL,
    plano TEXT DEFAULT 'trial',
    data_inicio_trial DATETIME DEFAULT CURRENT_TIMESTAMP,
    data_expiracao DATETIME,
    status TEXT DEFAULT 'ativo',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
  )`);

    mainDb.run(`CREATE TABLE IF NOT EXISTS pagamentos (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    barbearia_id INTEGER NOT NULL,
    valor REAL NOT NULL,
    plano TEXT NOT NULL,
    data_pagamento DATETIME DEFAULT CURRENT_TIMESTAMP,
    data_expiracao DATETIME NOT NULL,
    status TEXT DEFAULT 'pago',
    FOREIGN KEY(barbearia_id) REFERENCES barbearias(id)
  )`);
});

// Criar banco de dados para cada barbearia
function criarBancoBarbearia(barbeariaId) {
    const dbPath = `barbearia_${barbeariaId}.db`;
    const db = new sqlite3.Database(dbPath);

    db.serialize(() => {
        db.run(`CREATE TABLE IF NOT EXISTS agendamentos (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      nome TEXT NOT NULL,
      servico TEXT NOT NULL,
      preco REAL NOT NULL,
      data TEXT NOT NULL,
      hora TEXT NOT NULL,
      telefone TEXT,
      created_at DATETIME DEFAULT CURRENT_TIMESTAMP
    )`);

        db.run(`CREATE TABLE IF NOT EXISTS clientes (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      nome TEXT NOT NULL,
      email TEXT,
      telefone TEXT,
      data_cadastro DATETIME DEFAULT CURRENT_TIMESTAMP
    )`);

        db.run(`CREATE TABLE IF NOT EXISTS servicos (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      nome TEXT NOT NULL,
      descricao TEXT,
      preco REAL NOT NULL,
      duracao INTEGER DEFAULT 30
    )`);

        db.run(`CREATE TABLE IF NOT EXISTS config (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      nome_empresa TEXT DEFAULT 'Minha Barbearia',
      telefone TEXT DEFAULT '',
      endereco TEXT DEFAULT '',
      horario_funcionamento TEXT DEFAULT 'Seg-Sex: 09h às 20h',
      logo_texto TEXT DEFAULT '✂️'
    )`);

        // Inserir serviços padrão
        db.get("SELECT COUNT(*) as total FROM servicos", (err, row) => {
            if (row && row.total === 0) {
                const servicos = [
                    ['Corte de Cabelo', 'Corte tradicional', 35.00, 30],
                    ['Barba', 'Barba completa', 25.00, 30],
                    ['Corte + Barba', 'Pacote completo', 55.00, 60],
                    ['Pezinho', 'Hidratação e corte', 20.00, 20],
                    ['Platinado', 'Descoloração', 120.00, 120]
                ];
                servicos.forEach(s => {
                    db.run('INSERT INTO servicos (nome, descricao, preco, duracao) VALUES (?, ?, ?, ?)', s);
                });
            }
        });

        db.run(`INSERT OR IGNORE INTO config (id, nome_empresa) VALUES (1, 'Minha Barbearia')`);
    });

    return db;
}

// Cadastrar nova barbearia (trial)
app.post('/api/cadastrar-barbearia', (req, res) => {
    const { nome, email, senha, telefone, cnpj } = req.body;

    if (!nome || !email || !senha) {
        return res.status(400).json({ erro: 'Nome, email e senha são obrigatórios!' });
    }

    mainDb.get('SELECT * FROM barbearias WHERE email = ?', [email], (err, barbeariaExistente) => {
        if (err) {
            return res.status(500).json({ erro: err.message });
        }

        if (barbeariaExistente) {
            return res.status(400).json({ erro: 'Este email já está cadastrado!' });
        }

        const dataExpiracao = new Date();
        dataExpiracao.setDate(dataExpiracao.getDate() + 14);

        mainDb.run(
            'INSERT INTO barbearias (nome, email, senha, telefone, cnpj, plano, data_expiracao) VALUES (?, ?, ?, ?, ?, ?, ?)',
            [nome, email, senha, telefone || '', cnpj || '', 'trial', dataExpiracao.toISOString()],
            function (err) {
                if (err) {
                    return res.status(500).json({ erro: err.message });
                }

                const barbeariaId = this.lastID;
                criarBancoBarbearia(barbeariaId);

                res.json({
                    id: barbeariaId,
                    mensagem: '✅ Barbearia cadastrada com sucesso! Trial de 14 dias.',
                    data_expiracao: dataExpiracao.toISOString()
                });
            }
        );
    });
});

// Login da barbearia
app.post('/api/login-barbearia', (req, res) => {
    const { email, senha } = req.body;

    mainDb.get('SELECT * FROM barbearias WHERE email = ? AND senha = ?', [email, senha], (err, barbearia) => {
        if (err) {
            return res.status(500).json({ erro: err.message });
        }

        if (!barbearia) {
            return res.status(401).json({ erro: 'Email ou senha inválidos!' });
        }

        const hoje = new Date();
        const expiracao = new Date(barbearia.data_expiracao);

        if (barbearia.status === 'inativo') {
            return res.status(403).json({ erro: 'Sua conta está desativada. Entre em contato com o suporte.' });
        }

        if (expiracao < hoje && barbearia.plano === 'trial') {
            return res.status(403).json({
                erro: 'Seu período de teste expirou! Faça um plano para continuar.',
                expirado: true
            });
        }

        const token = crypto.randomBytes(32).toString('hex');

        res.json({
            id: barbearia.id,
            nome: barbearia.nome,
            email: barbearia.email,
            plano: barbearia.plano,
            data_expiracao: barbearia.data_expiracao,
            token: token,
            dias_restantes: Math.ceil((expiracao - hoje) / (1000 * 60 * 60 * 24))
        });
    });
});

// ============= MIDDLEWARE =============

function verificarAcesso(req, res, next) {
    const barbeariaId = req.headers['barbearia-id'];
    const token = req.headers['authorization'];

    if (!barbeariaId || !token) {
        return res.status(401).json({ erro: 'Não autorizado' });
    }

    mainDb.get('SELECT * FROM barbearias WHERE id = ?', [barbeariaId], (err, barbearia) => {
        if (err || !barbearia) {
            return res.status(401).json({ erro: 'Barbearia não encontrada' });
        }

        if (barbearia.status !== 'ativo') {
            return res.status(403).json({ erro: 'Conta desativada' });
        }

        const hoje = new Date();
        const expiracao = new Date(barbearia.data_expiracao);

        if (expiracao < hoje && barbearia.plano === 'trial') {
            return res.status(403).json({ erro: 'Período de teste expirado', expirado: true });
        }

        req.barbearia = barbearia;
        next();
    });
}

function getBarbeariaDb(barbeariaId) {
    return new sqlite3.Database(`barbearia_${barbeariaId}.db`);
}

app.get('/api/dashboard', verificarAcesso, (req, res) => {
    const db = getBarbeariaDb(req.barbearia.id);
    const hoje = new Date().toISOString().split('T')[0];
    const semana = [];

    for (let i = 0; i < 7; i++) {
        const data = new Date();
        data.setDate(data.getDate() + i);
        semana.push(data.toISOString().split('T')[0]);
    }

    const query = `SELECT * FROM agendamentos WHERE data IN (${semana.map(() => '?').join(',')}) ORDER BY data, hora`;

    db.all(query, semana, (err, agendamentos) => {
        if (err) {
            db.close();
            return res.status(500).json({ erro: err.message });
        }

        const faturamentoDia = agendamentos
            .filter(a => a.data === hoje)
            .reduce((sum, a) => sum + (a.preco || 0), 0);

        const faturamentoSemana = agendamentos.reduce((sum, a) => sum + (a.preco || 0), 0);

        const agendamentosPorDia = {};
        semana.forEach(dia => {
            agendamentosPorDia[dia] = agendamentos.filter(a => a.data === dia);
        });

        res.json({
            semana: semana,
            agendamentos: agendamentosPorDia,
            faturamento: {
                dia: faturamentoDia,
                semana: faturamentoSemana
            },
            totalAgendamentos: agendamentos.length
        });

        db.close();
    });
});

    const mes = agora.getMonth() + 1;
    const mesStr = `${ano}-${mes.toString().padStart(2, '0')}%`;

    db.all('SELECT * FROM agendamentos WHERE data LIKE ?', [mesStr], (err, agendamentos) => {
        db.close();
        if (err) {
            return res.status(500).json({ erro: err.message });
        }

        const faturamentoMes = agendamentos.reduce((sum, a) => sum + (a.preco || 0), 0);
        res.json({ faturamento: faturamentoMes, total: agendamentos.length });
    });
});

app.post('/api/renovar-assinatura', verificarAcesso, (req, res) => {
    const { plano_id, valor } = req.body;
    const barbearia = req.barbearia;

    const hoje = new Date();
    let novaExpiracao = new Date();

    if (plano_id === 'mensal') novaExpiracao.setDate(hoje.getDate() + 30);
    else if (plano_id === 'trimestral') novaExpiracao.setDate(hoje.getDate() + 90);
    else if (plano_id === 'anual') novaExpiracao.setDate(hoje.getDate() + 365);

    mainDb.run(
        `INSERT INTO pagamentos (barbearia_id, valor, plano, data_expiracao) VALUES (?, ?, ?, ?)`,
        [barbearia.id, valor, plano_id, novaExpiracao.toISOString()],
        (err) => {
            if (err) {
                return res.status(500).json({ erro: err.message });
            }

            mainDb.run(
                `UPDATE barbearias SET plano = ?, data_expiracao = ?, status = 'ativo' WHERE id = ?`,
                [plano_id, novaExpiracao.toISOString(), barbearia.id],
                (err) => {
                    if (err) {
                        return res.status(500).json({ erro: err.message });
                    }

                    res.json({
                        mensagem: '✅ Assinatura renovada com sucesso!',
                        data_expiracao: novaExpiracao.toISOString(),
                        dias: plano_id === 'mensal' ? 30 : plano_id === 'trimestral' ? 90 : 365
                    });
                }
            );
        }
    );

        } else if (this.changes === 0) {
            res.status(404).json({ erro: 'Serviço não encontrado' });
        } else {
            res.json({ mensagem: '✅ Serviço excluído com sucesso!' });
        }
    });
});
